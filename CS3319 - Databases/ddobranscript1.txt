SHOW databases;
DROP DATABASE IF EXISTS vetdb;
CREATE DATABASE vetdb;
USE vetdb;
SHOW TABLES;

CREATE TABLE TA (
    firstname VARCHAR(30) NOT NULL,
    lastname VARCHAR(30) NOT NULL,
    studentNumber VARCHAR(9),
    westernID VARCHAR(8),
    isPhD BOOLEAN,
    picture VARCHAR(150),
    UNIQUE(westernID),
    PRIMARY KEY (westernID)
    FOREIGN KEY (westernID) REFERENCES Instructors(westernID)
);

CREATE TABLE Course (
	courseNumber VARCHAR(6),
	courseName VARCHAR(30),
	UNIQUE(courseNumber),
	PRIMARY KEY (courseNumber)
);

CREATE TABLE Instructors (
	firstName VARCHAR(30) NOT NULL,
	lastName VARCHAR(30) NOT NULL,
	westernID VARCHAR(8) NOT NULL,
	UNIQUE(westernID),
	PRIMARY KEY (westernID)
);
