SHOW databases;
DROP DATABASE IF EXISTS ddobrandb;
CREATE DATABASE ddobrandb;
USE ddobrandb;
SHOW TABLES;

CREATE TABLE TAs (
    firstname VARCHAR(30) NOT NULL,
    lastname VARCHAR(30) NOT NULL,
    studentNumber VARCHAR(9),
    westernID VARCHAR(8),
    type VARCHAR(7),
    picture VARCHAR(150),
    UNIQUE(westernID),
    PRIMARY KEY (westernID)
    FOREIGN KEY (headSupervisorWesternId)
    	REFERENCES Instructors(westernID)
);

CREATE TABLE Courses (
	courseNumber VARCHAR(6),
	courseName VARCHAR(30),
	UNIQUE(courseNumber),
	PRIMARY KEY (courseNumber)
);

CREATE TABLE Instructors (
	firstName VARCHAR(30) NOT NULL,
	lastName VARCHAR(30) NOT NULL,
	westernID VARCHAR(8) NOT NULL,
	UNIQUE(westernID),
	PRIMARY KEY (westernID)
);

CREATE TABLE CoSupervisors (
	FOREIGN KEY (instructorId)
		REFERENCES Instructors(westernId),
	FOREIGN KEY (taId)
		REFERENCES TA(westernID),
	PRIMARY KEY (instructorId, taId)
);

CREATE TABLE CourseInfo (
	FOREIGN KEY (courseNumber) REFERENCES Courses(courseNumber),
	term VARCHAR(6),
	year INT,
	studentsEnrolled INT,
	PRIMARY KEY (courseNumber, term, year)
);

CREATE TABLE TAassignedToCourse (
	FOREIGN KEY (taId) REFERENCES TAs(westernID),
	FOREIGN KEY (courseNumber) REFERENCES CourseInfo(courseNumber)
	PRIMARY KEY (taId, courseNumber)
);

SHOW TABLES;
